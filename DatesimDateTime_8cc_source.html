41 secTy DatasimDateTime::secs(long hours, long minutes, long secos)
42 /*
43  * return the number of seconds after midnight (12:00 AM) from
44  * hours, minutes and seconds. If one of them is < 0 then
45  * return 0. (=midnight). The number of hours, minutes and seconds
46  * may be any value (> 0).
47  */
48 {
49  if (hours < 0 OR minutes < 0 OR secos < 0)
50  return(0);
51  else {
52  minutes += (secos / 60); // #minutes in seconds
53  secos %= 60; // rest seconds
54  hours += (minutes / 60); // #hours in minutes
55  minutes %= 60; // rest minutes
56  hours %= 24; // discard day-limit
57  return(hours * 60l * 60l + minutes * 60l + secos);// total
58  // number of seconds after 12:00
59  }
60 }
