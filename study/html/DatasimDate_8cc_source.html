69 int DatasimDate::day_test(int day, int month, int year) const
70 /*
71  * Test if the given day (which is a month-day-number) is a valid
72  * day in the given month and year. It checks for leapyears, etc.
73  * If it is valid, return the given day, else return 0.
74  */
75 {
76  static int numdays[] = // number of days in month
77  { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
78 
79  if (day < 1 OR month < 1 OR month > 12 OR year == 0)
80  return(0);
81  else if (month == 2)
82  if ((year % 4 != 0) OR (year % 100 == 0 AND year % 400 != 0))
83  return(day <= 28 ? day : 0);
84  else
85  return(day <= 29 ? day : 0);
86  else
87  return(day <= numdays[month - 1] ? day : 0);
88 }

154 julTy DatasimDate::j_days(julTy day, julTy month, julTy year) const
155 /*
156  * return the number of Julian days from the given Gregorian DatasimDate
157  * ( in the variables day, month and year). If a wrong DatasimDate is given,
158  * it will return 0. Algorithm 199 from Communications
159  * af the ACM Volume6, No.8, (Aug. 1963) p.144. Gregorian calender
160  * started September 14, 1752. The algorithm is only valid for
161  * DatasimDates from this calender, other DatasimDates may result in invalid Julian
162  * days.
163  */
164 {
165  julTy c, ya;
166 
167  if (day == 0 OR month == 0 OR year == 0) // wrong DatasimDate.
168  return(0);
169  else { // valid DatasimDate, algorithm to convert to Julian days.
170  if (month > 2)
171  month -= 3;
172  else {
173  month += 9;
174  year--;
175  }
176  c = year / 100;
177  ya = year - 100 * c;
178  return(((146097 * c) >> 2) + ((1461 * ya) >> 2) +
179  (153 * month + 2) / 5 + day + 1721119);
180  }
181 }

183 void DatasimDate::jul_to_greg(julTy& day, julTy& month, julTy& year) const
184 /*
185  * get the Gregorian DatasimDate (returned in the referenced variables
186  * day, month and year) from the number of Julian days (member
187  * Julian_days) from this DatasimDate. Algorithm 199 from Communications
188  * af the ACM Volume6, No.8, (Aug. 1963) p.144. If the number of
189  * Julian days = 0, return January 1, 1900. Gregorian calender started
190  * September 14, 1752.
191  */
192 {
193  if (Julian_days == 0) { // return January 1, 1900
194  day = 1; month = 1; year = 1900;
195  return;
196  }
197  else { // This is the algorithm.
198  julTy j = Julian_days - 1721119l;
199  year = (julTy)(((j << 2) - 1) / 146097l);
200  j = (j << 2) - 1 - 146097l * year;
201  day = (julTy)(j >> 2);
202  j = ((day << 2) + 3) / 1461;
203  day = (julTy)((day << 2) + 3 - 1461 * j);
204  day = (day + 4) >> 2;
205  month = (5 * day - 3) / 153;
206  day = 5 * day - 3 - 153 * month;
207  day = (day + 5) / 5;
208  year = (julTy)(100 * year + j);
209  if (month < 10)
210  month += 3;
211  else {
212  month -=9;
213  year++;
214  }
215  }
216 }

409 DatasimDate DatasimDate::add_period(const julTy& days, const julTy& months, const julTy& years) const
410 /*
411  * add a period of time to this (which itself will not be changed) DatasimDate
412  * and return that DatasimDate. The period of time can be any number of days,
413  * months and years. First the number of years is added (if specified)
414  * then the number of months (if specified) and finaly the number of
415  * days. So:
416  * DatasimDate d = DatasimDate("28-2-1980").add_period(3, 13, 1);
417  * results in March 31, 1982.
418  */
419 {
420  julTy nday, nmonth, nyear;
421 
422  jul_to_greg(nday, nmonth, nyear); // calc Greg. cal. from This DatasimDate
423  nmonth += months; // add months
424  nyear += ((nmonth - 1) / 12); // how many years from months ?
425  nmonth = (nmonth - 1) % 12 + 1; // remaining months (1..12)
426  nyear += years; // add years
427  while (day_test(nday, nmonth, nyear) == 0) // day still valid in month?
428  nday--; // no, subtract one monthday
429 
430  julTy total_days = days + j_days(nday, nmonth, nyear); // total Julian days
431  DatasimDate d(total_days); // construct new DatasimDate
432  return(d); // return new DatasimDate
433 }

435 DatasimDate DatasimDate::sub_period(julTy days, julTy months, julTy years) const
436 /*
437  * subtract a period of time from this (which will not be changed) DatasimDate
438  * and return that DatasimDate. The period of time can be any number of days,
439  * months and years. First the number of years is subtracted (if speci-
440  * fied) then the number of months (if specified) and finaly the number of
441  * days. So:
442  * DatasimDate d = DatasimDate("1-3-1980").sub_period(3, 13, 1);
443  * results in January 29, 1978.
444  */
445 {
446  julTy nday, nmonth, nyear;
447 
448  jul_to_greg(nday, nmonth, nyear); // calc Greg cal. from this DatasimDate
449  nyear -= (months / 12); // how many years in months ?
450  months %= 12; // remaining months;
451  if (months >= nmonth) { // more than months in this DatasimDate
452  nyear--; // one year earlier
453  nmonth = 12 + nmonth - months; // remaining months
454  }
455  else nmonth -= months;
456  nyear -= years;
457  while (day_test(nday, nmonth, nyear) == 0) // day still valid in month?
458  nday--; // no, subtract one monthday
459 
460  julTy total_days = j_days(nday, nmonth, nyear) - days;
461  DatasimDate d(total_days);
462  return(d);
463 }
