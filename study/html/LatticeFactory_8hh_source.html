40  virtual void updateLattice
41  (Lattice<double, int, 2>& source, double rootValue) const
42  { // Find the depth of the lattice; this a Template Method Pattern
43 
44  int si = source.MinIndex();
45  source[si][source[si].MinIndex()] = rootValue;
46 
47  // Loop from the min index to the end index
48  for (int j = source.MinIndex() + 1; j <= source.MaxIndex(); j++)
49  {
50  for (int i = source[j].MinIndex(); i < source[j].MaxIndex(); i++)
51  {
52  source[j][i] = d * source[j-1][i];
53  source[j][i+1] = u * source[j-1][i];
54  }

56  }
