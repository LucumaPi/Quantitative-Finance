34 class BasketStrategy
35 { // 2-asset basket option payoff
36 
37 private:
38  double K; // Strike
39  double w; // +1 call, -1 put
40  double w1, w2; // w1 + w2 = 1
41 public:
42  // All classes need default constructor
43  BasketStrategy()
44  { K = 95.0; w = +1; w1 = 0.5; w2 = 0.5;
45  }
46 
47  BasketStrategy(double strike, double cp,double weight1, double weight2)
48  { K = strike; w = cp; w1 = weight1; w2 = weight2;
49  }
50  double payoff(double S1, double S2) const
51  {
52 
53  double sum = w1*S1 + w2*S2;
54 
55  if (w*(sum-K) > 0.0)
56  return (w*(sum - K));
57  return 0.0;
58  }
