189 template <class V, class I> NumericMatrix<V,I> createMatrix(
190  const Vector<V,I>& lower, const Vector<V,I>& diagonal, const Vector<V,I>& upper)
191 { // Create a matrix based on the diagonal elements of a tridiagonal matrix; bit wasteful of
192  // space but who cares! I've got a Pentium with a giga of memory
193 
194 
195  // We use the indexing as with LU decomposition routines, i.e.
196  //
197  // lower [2, J]
198  // diagonal [1, J]
199  // upper [1, J-1]
200  //
201 
202  NumericMatrix<V, I> result(diagonal.Size(), diagonal.Size(), 1, 1);
203 
204  // All values set to 0
205  initMatrix(result, V(0.0));
206 
207  // Initialise the 'extremities' of the matrix
208  result(result.MaxRowIndex(), result.MaxColumnIndex()) = diagonal[diagonal.MaxIndex()];
209  result(result.MaxRowIndex(), result.MaxColumnIndex() -1) = lower[lower.MaxIndex()];
210 
211  result(1, 1) = diagonal[diagonal.MinIndex()];
212  result(1, 2) = upper[upper.MinIndex()];
213 
214  // Now the 'uniform' part of the matrix
215  for (I i = result.MinRowIndex() + 1; i <= result.MaxRowIndex() - 1; i++)
216  {
217 
218  // Create the diagonal element
219  result(i, i-1) = lower[i];
220  result(i, i) = diagonal[i];
221  result(i, i+1) = upper[i];
222  }
223 
224  return result;
225 }

420 template <class V, class I> V dominantEigenvalue(const NumericMatrix<V,I>& A, const Vector<V,I>& svec,
421  const V& tolerance)
422 {
423 
424  Vector<V,I> startV = svec;
425  Vector<V,I> currVector = (A * svec);
426  V tmp(0.0);
427  V nextValue (tmp);
428 
429  L1:
430  tmp = RayleighQuotient(A, currVector);
431 
432  if (fabs(tmp-nextValue) > tolerance)
433  {
434  nextValue = tmp;
435  startV = currVector;
436  currVector = A * startV;
437  goto L1;
438  }
439 
440  return nextValue;
441
